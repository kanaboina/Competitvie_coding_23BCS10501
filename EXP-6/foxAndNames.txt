bool dfs(int&i,vector<vector<int>>&adj,vector<int>&vis,vector<int>&path) {
    vis[i]=1;
    path[i]=1;
    for(auto x:adj[i]) {if(!vis[x]){if(dfs(x,adj,vis,path)) return true;}
    else if(path[x]==1) return true;
    }path[i]=0;
    return false;
}

     for(int i=0;i<26;i++) if(!vis[i]||flag)if(dfs(i,adj,vis,path)||flag) {cout<<"Impossible"<<endl; return;}

    for(int i=0;i<26;i++) for(auto&j:adj[i]){vis[i]++;vis[j]++; indeg[j]++;}

 q.pop();
